矩形自定义长和宽，求面积和周长

public class Matrix {
public double chang;
public double kuan;

Matrix(double chang,double kuan)
{
    this.chang=chang;
    this.kuan=kuan;
}
public void s()
{
    double ss;
    ss=chang*kuan;
    System.out.println("矩形的面积是："+ss);
    double c=(chang+kuan)*2;
    System.out.println("矩形的周长是："+c);
}


public static void main(String[] args) {
    // TODO Auto-generated method stub
  Matrix h=new Matrix(2,3);
  h.s();
}

}
